    class SalesPersonInterface {
        + selectFoodItems()
        + selectOccasions()
        + viewRecommendations()
        + retryWithNewParameters()
    }
    class SystemInterface {
        + promptSelection()
        + displayRecommendations()
        + notifyNoWineAvailable()
    }
}

package Control {
    class WineRecommendationEngine {
        + filterWines(foodItems, occasions)
        + checkWineAvailability()
    }
}

    class FoodItem {
        - name: String
    }
    class Occasion {
        - name: String
    }
    class Wine {
        - name: String
        - relevanceScore: int
    }
}
